Two click-handlers

const button = document.querySelector("button");

button.addEventListener("click", () => {
    console.log("1");
    queueMicrotask(() => {
        console.log("Microtask 1");
    });
    setTimeout(() => {
        console.log("Macrotask 1");
    }, 0);
    requestAnimationFrame(() => {
        console.log("RAF 1");
    });
});

button.addEventListener("click", () => {
    console.log("2");
    queueMicrotask(() => {
        console.log("Microtask 2");
    });
    setTimeout(() => {
        console.log("Macrotask 2");
    }, 0);
    requestAnimationFrame(() => {
        console.log("RAF 2");
    });
});

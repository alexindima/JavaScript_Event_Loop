    function runExample() {
        setTimeout(() => {
            const div1 = document.createElement("div");
            div1.className = "box";
            document.body.appendChild(div1);

            const syncRect1 = div1.getBoundingClientRect();
            console.log("Sync 1:", JSON.stringify(syncRect1));

            const div2 = document.createElement("div");
            div2.className = "box";
            document.body.appendChild(div2);

            const syncRect2 = div2.getBoundingClientRect();
            console.log("Sync 2:", JSON.stringify(syncRect2));

            const div3 = document.createElement("div");
            div3.className = "box";
            document.body.appendChild(div3);

            requestAnimationFrame(() => {
                console.log("rAF");
            });
        }, 2000);
    }
